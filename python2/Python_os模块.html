    1.os模块中常用的方法：

    import os

#获取当前文件所在的文件夹路径
print(os.getcwd())

#获取当前文件夹下面的所有文件
print(os.listdir('E:\\projects\\Python\\testFile'))

#新建一个文件夹
os.mkdir('test')

#删除一个文件夹
os.rmdir('test')

#重命名
os.rename('apple.py','apple1.py')
