    import time
def buy(name):                      # 消费者
    print('%s上街去买蛋' %name)
    while True:
        eggs=yield
        print('%s买了%s' %(name,eggs))


def product_eggs():              # 生产者
    a =buy('小明')
    a.__next__()
    for i  in range(1,11):
        time.sleep(1)
        a.send('%s个鸡蛋' %i)

product_eggs()
