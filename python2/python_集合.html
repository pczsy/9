    1.add添加

    a = {1,2,3}
# 向集合a中添加'wdc'
a.add('wdc')
print(a)

    2.discard删除

    a = {1,2,'wdc','yhf'}
# 删除集合a中的数字2，集合是无序的，不能通过索引删除
a.discard(2)
print(a)

    3.update批量添加

    a = {1,2}
# 向集合中批量添加元素
a.update({'wdc','yhf','qwdf'})
print(a)

    4.intersection取交集

    a = {1,2,'wdc'}
b = {2,3,'wdc'}
# 取集合a和集合b的交集并重新赋值给c
c = a.intersection(b)
print(c)

    5.union取并集

    a = {1,2,'wdc'}
b = {2,3,'wdc'}
# 取集合a和集合b的并集并重新赋值给c
c = a.union(b)
print(c)

    6.difference取差集

    a = {1,2,'wdc'}
b = {2,3,'wdc'}
# 取集合a中有，集合b中没有的，并赋值给c（差集）
c = a.difference(b)
print(c)

    7.symmetric_difference对称差集

    a = {1,2,'wdc'}
b = {2,3,'wdc'}
# 集合a和集合b互相一次差集，结果放在一起的集合赋值给c
c = a.symmetric_difference(b)
print(c)
